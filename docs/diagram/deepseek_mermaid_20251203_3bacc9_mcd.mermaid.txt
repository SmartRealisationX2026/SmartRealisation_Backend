    namespace SystèmeMediLink {
        class Utilisateur {
            +UUID id
            +String email
            +Rôle rôle
            +Profil profil
            +Préférences préférences
        }

        class Adresse {
            +Ville ville
            +Arrondissement arrondissement
            +String rue
            +PointGPS coordonnées
        }

        class Médicament {
            +String nomCommercial
            +DCI dénominationCommune
            +Catégorie catégorie
            +Forme forme
            +Dosage dosageRéférence
            +bool prescriptionObligatoire
        }

        class Pharmacie {
            +String nom
            +Adresse adresse
            +Horaires horaires
            +Contact contact
            +StatusÉtablissement statut
            +[Inventaire] stocks
        }

        class ItemInventaire {
            +Médicament médicament
            +String lot
            +Date expiration
            +Quantité quantité
            +Prix prix
            +bool disponible
        }

        class Recherche {
            +Médicament médicamentCible
            +Localisation localisation
            +Filtres filtres
            +[Pharmacie] résultats
            +DateHeure dateRecherche
        }

        class SystèmeAnalytique {
            +DashboardAdmin dashboard
            +HeatmapCartographique heatmap
            +RapportsMétier rapports
            +AlertesSystème alertes
        }
    }

    %% Valeurs et Entités de Référence
    class Rôle {
        <<enumération>>
        PATIENT
        PHARMACIEN
        ADMINISTRATEUR
    }

    class Catégorie {
        +String code
        +String nomFr
        +String nomEn
        +Catégorie parent
    }

    class Forme {
        +String code
        +String nomFr
        +String nomEn
    }

    class DCI {
        +String codeATC
        +String nomInternational
        +String classeThérapeutique
    }

    class PointGPS {
        +Decimal latitude
        +Decimal longitude
    }

    class Quantité {
        +int valeur
        +String unité
    }

    class Prix {
        +Decimal montant
        +String devise "FCFA"
    }

    %% Relations
    Utilisateur "1" -- "*" Recherche : effectue
    Utilisateur "1" -- "0..*" Pharmacie : possède
    Médicament "*" -- "1" Catégorie : appartient_à
    Médicament "*" -- "1" Forme : a_pour_forme
    Pharmacie "1" -- "*" ItemInventaire : détient
    ItemInventaire "*" -- "1" Médicament : référence
    Recherche "1" -- "1" Médicament : cible
    Recherche "1" -- "*" Pharmacie : retourne
    SystèmeAnalytique "1" -- "*" Recherche : analyse
    Pharmacie "1" -- "1" Adresse : située_à
